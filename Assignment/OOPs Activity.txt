TASK: Create a Passenger class with basic attributes (e.g., name, age, ticket number)
TASK: Define a simple method to display the passenger's details.
ANS: 
#include <iostream>
#include <string>
using namespace std;
class Passenger {
private:
    string name;
    int age;
    string ticketNumber;
public:
    Passenger(string passengerName, int passengerAge, string passengerTicketNumber) {
        name = passengerName;
        age = passengerAge;
        ticketNumber = passengerTicketNumber;
    }
    void displayDetails() {
        cout << "Passenger Details:" << endl;
        cout << "Name: " << name << endl;
        cout << "Age: " << age << endl;
        cout << "Ticket Number: " << ticketNumber << endl;
    }
};
int main() {
    Passenger passenger1("Ishant", 19, "12356");
    passenger1.displayDetails();
    return 0;
}
